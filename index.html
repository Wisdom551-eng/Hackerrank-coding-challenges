vector compareTriplets(vector a, vector b) { int alice_score = 0; int bob_score = 0; for (int i = 0; i < 3; i++) { // Since both `a` and `b` have exactly 3 elements if (a[i] > b[i]) { alice_score++; } else if (a[i] < b[i]) { bob_score++; } } return {alice_score, bob_score}; // Return the scores as a vector }
